USE Northwind


SELECT 
  e.FirstName,
  tab1.OrderID, 
  tab1.OrderDate
FROM Employees e
cross apply 
      (select top(3) * from  Orders where e.EmployeeID = orders.EmployeeID order by orderdate desc) as tab1 on
e.EmployeeID = tab1.EmployeeID
order by e.FirstName
